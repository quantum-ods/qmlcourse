def quanv(image):
    """–§—É–Ω–∫—Ü–∏—è –∫–≤–∞–Ω—Ç–æ–≤–æ–π —Å–≤–µ—Ä—Ç–∫–∏ –Ω–∞–¥ –≤—Ö–æ–¥–Ω—ã–º –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ–º."""

    out = np.zeros((14, 14, n_wires))

    # –¶–∏–∫–ª—ã –ø–æ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞–º –≤–µ—Ä—Ö–Ω–µ–≥–æ –ª–µ–≤–æ–≥–æ –ø–∏–∫—Å–µ–ª—è –±–ª–æ–∫–æ–≤ 2—Ö2
    for j in range(0, 28, 2):
        for k in range(0, 28, 2):
            # –û–±—Ä–∞–±–æ—Ç–∫–∞ –±–ª–æ–∫–∞ 2x2 –∏–∑ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –∫–≤–∞–Ω—Ç–æ–≤–æ–π —Ü–µ–ø—å—é
            q_results = circuit(
                [
                    image[j, k, 0],
                    image[j, k + 1, 0],
                    image[j + 1, k, 0],
                    image[j + 1, k + 1, 0]
                ]
            )
            # –ó–∞–ø–∏—Å—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ –Ω–∞–±–ª—é–¥–µ–Ω–∏—è –≤ –≤—ã—Ö–æ–¥–Ω–æ–π –ø–∏–∫—Å–µ–ª—å (j/2, k/2)
            for c in range(n_wires):
                out[j // 2, k // 2, c] = q_results[c]
    return out
